Generate 1000 draws form the poisson distribution for values of lambda

lambda_values <- c(1, 5, 10, 50, 75)
data <- data.frame()

for(x in lambda_values) {
  draws <- rpois(1000, lambda = x)
  data <- rbind(data, data.frame(lambda = x, value = draws))
}
